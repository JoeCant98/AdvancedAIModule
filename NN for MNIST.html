#import required libraries
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout, Activation, Flatten, Conv2D, MaxPooling2D

#Save the mnist dataset into local variable mnist
#from source [1] 
mnist = tf.keras.datasets.mnist

#initiate some variables for use later on
img_rows, img_cols = 28, 28
batch_size=128
image_vector_size = 28*28
num_category = 10
image_size = 784 # 28*28
num_classes = 10 # ten unique digits

#Load mnist data into variables below
(x_train, y_train), (x_test, y_test) = mnist.load_data()

#Reshape the data to a 2D space
x_train = x_train.reshape(x_train.shape[0], image_vector_size)

x_test = x_test.reshape(x_test.shape[0], image_vector_size)

#convert class vectors to binary class matrices
y_train = tf.keras.utils.to_categorical(y_train, num_category)
y_test = tf.keras.utils.to_categorical(y_test, num_category)
#end of source [1] https://medium.com/tebs-lab/how-to-classify-mnist-digits-with-different-neural-network-architectures-39c75a0f03e3

#initialize and build model
model = Sequential()

#Fully connected dense layers converting input to size (*,64) at the end.
#First layer requires the input shape
model.add(Dense(32, activation='selu', input_shape=(image_size,)))
model.add(Dense(64, activation='selu'))

#Dropout function for convergence sake.
model.add(Dropout(0.25))

#This allows 10 outputs, from 0-9
model.add(Dense(num_classes, activation='softmax'))

#Compile
model.compile(loss='categorical_crossentropy',
              optimizer='adadelta',
              metrics=['accuracy'])

#Train model
model.fit(x_train, y_train,
          batch_size=batch_size,
          epochs=15,
          validation_data=(x_test, y_test))

#Evaluate model          
score = model.evaluate(x_test, y_test, verbose=0)
print('Test loss:', score[0])
print('Test accuracy:', score[1])

#import required libraries
import matplotlib.pyplot as plt
import tensorflow as tf

#Store History object in variable
result = model.fit(x_train, y_train,
          batch_size=batch_size,
          epochs=3,
          validation_data=(x_test, y_test))

#Use validation split
result = model.fit(x_train, y_train,
          batch_size=batch_size,
          epochs=3,
          validation_split=0.2)

#Save accuracy and loss values as variables
train_accu=result.history['accuracy']
val_accu=result.history['val_accuracy']
train_loss=result.history['loss']
val_loss=result.history['val_loss']

#plot training and validation accuracy graph
#from source[2]
plt.plot(train_accu)
plt.plot(val_accu)

plt.title('Model accuracy')

plt.xlabel('Epoch Number')
plt.legend(['Train', 'Test'], loc='upper right')
plt.show()
